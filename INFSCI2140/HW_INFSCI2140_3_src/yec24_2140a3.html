Analysis of Dirichlet prior parameter

To see if there’s a relationship between the smoothing scores and prior parameter \(\mu\).

First load text file to data frame:

library('ggplot2')
library('reshape2')

txt_list = list.files(pattern = "hw3main_output_mu=.*\\.txt$")
data = sapply(txt_list, function(x) {as.numeric(read.delim(file = x, header = F, sep =" ")$V5)})

mu = seq(500, 5000, by = 500)
query1 = data.frame(cbind(mu, t(data[2:21, ])))
query2 = data.frame(cbind(mu, t(data[24:43, ])))
query3 = data.frame(cbind(mu, t(data[45:64, ])))
query4 = data.frame(cbind(mu, t(data[66:85, ])))

Each query has 20 documents, each document has 10 scores based on different \(\mu\).

Plot \(\mu\) v.s. scores:

melted1 = melt(query1, id = "mu")
ggplot(melted1, aes(mu, value, color = variable)) + 
  geom_line() + geom_point(data = melted1) + ggtitle("Query 1") + 
  theme(plot.title = element_text(hjust = 0.5), legend.position = "none")

melted2 = melt(query2, id = "mu")
ggplot(melted2, aes(mu, value, color = variable)) + 
  geom_line() + geom_point(data = melted2) + ggtitle("Query 2") +
